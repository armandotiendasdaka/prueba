select c.nombre as nombre_cliente, v.nombre as nombre_vendedor
from febeca.cliente_vendedor cv
inner join febeca.cliente c on cv.cliente = c.cliente 
inner join febeca.vendedor v on cv.vendedor = v.vendedor 

-- 12

from febeca.articulo ar
inner join febeca.existencia_bodega ex
on ar.articulo = ex.articulo
where ex.bodega = 'BO07'
order by ex.cant_disponible desc
